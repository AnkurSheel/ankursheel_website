<p>If you're transitioning from C# to Java, the task of mapping C# LINQ functions to Java Streams can seem challenging
While the core concepts bear similarities, unravelling the nuances of syntax and functionality is crucial for success.</p>
<p>In this blog post, we will explore the Java equivalents for C# LINQ functions.</p>
<h2 id="operations">Operations</h2>
<h2 id="select-map">Select / map</h2>
<p>Transforms each element of a sequence based on a given function.</p>
<h3 id="c">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var squaredNumbers = numbers.Select(x =&gt; x * x).ToList();
</code></pre>
<h3 id="java">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
List&lt;Integer&gt; squaredNumbers = numbers.stream()
                                      .map(x -&gt; x * x)
                                      .collect(Collectors.toList());
</code></pre>
<h3 id="output">Output</h3>
<pre><code>[1, 4, 9, 16, 25]
</code></pre>
<h2 id="where-filter">Where / filter</h2>
<p>Filters a sequence based on a specified condition.</p>
<h3 id="c-1">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var evenNumbers = numbers.Where(x =&gt; x % 2 == 0).ToList();
</code></pre>
<h3 id="java-1">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
List&lt;Integer&gt; evenNumbers = numbers.stream()
                                   .filter(x -&gt; x % 2 == 0)
                                   .collect(Collectors.toList());
</code></pre>
<h3 id="output-1">Output</h3>
<pre><code>[2, 4]
</code></pre>
<h2 id="orderby-sorted">OrderBy / sorted</h2>
<p>Sorts the elements of a sequence in ascending order based on a specified key.</p>
<h3 id="c-2">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 3, 1, 4, 2, 5 };
var sortedNumbers = numbers.OrderBy(x =&gt; x).ToList();
</code></pre>
<h3 id="java-2">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(3, 1, 4, 2, 5);
List&lt;Integer&gt; sortedNumbers = numbers.stream()
                                     .sorted()
                                     .collect(Collectors.toList());
</code></pre>
<h3 id="output-2">Output</h3>
<pre><code>[1, 2, 3, 4, 5]
</code></pre>
<h2 id="orderbydescending-sorted">OrderByDescending / sorted</h2>
<p>Sorts the elements of a sequence in descending order based on a specified key.</p>
<h3 id="c-3">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 5, 2, 9, 1, 7 };
var sortedNumbers = numbers.OrderByDescending(n =&gt; n);
</code></pre>
<h3 id="java-3">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(5, 2, 9, 1, 7);

List&lt;Integer&gt; sortedNumbers = numbers.stream()
                                     .sorted(Comparator.reverseOrder())
                                     .collect(Collectors.toList());
</code></pre>
<h3 id="output-3">Output</h3>
<pre><code>[9, 7, 5, 2, 1]
</code></pre>
<h2 id="groupby-groupingby">GroupBy / groupingBy</h2>
<p>Groups the elements of a sequence based on a specified key.</p>
<h3 id="c-4">C#</h3>
<pre><code class="language-csharp">var people = new List&lt;Person&gt;
{
    new Person { Name = &quot;Alice&quot;, Age = 25 },
    new Person { Name = &quot;Bob&quot;, Age = 30 },
    new Person { Name = &quot;Charlie&quot;, Age = 25 }
};
var peopleByAge = people.GroupBy(p =&gt; p.Age);
</code></pre>
<h3 id="java-4">Java</h3>
<pre><code class="language-java">List&lt;Person&gt; people = Arrays.asList(
        new Person(&quot;Alice&quot;, 25),
        new Person(&quot;Bob&quot;, 30),
        new Person(&quot;Charlie&quot;, 25)
);
Map&lt;Integer, List&lt;Person&gt;&gt; peopleByAge = people.stream()
                                               .collect(Collectors.groupingBy(Person::getAge));
</code></pre>
<h3 id="output-4">Output</h3>
<pre><code>{
    25: [{ Name: &quot;Alice&quot;, Age: 25 }, { Name: &quot;Charlie&quot;, Age: 25 }],
    30: [{ Name: &quot;Bob&quot;, Age: 30 }]
}
</code></pre>
<h2 id="join-join">Join / join</h2>
<p>Combines two sequences based on matching keys.</p>
<h3 id="c-5">C#</h3>
<pre><code class="language-csharp">var persons = new List&lt;Person&gt;
{
    new Person { Id = 1, Name = &quot;John&quot; },
    new Person { Id = 2, Name = &quot;Jane&quot; },
    new Person { Id = 3, Name = &quot;Alice&quot; }
};

var addresses = new List&lt;Address&gt;
{
    new Address { Id = 1, City = &quot;New York&quot; },
    new Address { Id = 2, City = &quot;London&quot; },
    new Address { Id = 3, City = &quot;Paris&quot; }
};

var joinedData = persons.Join(addresses,
                              person =&gt; person.Id,
                              address =&gt; address.Id,
                              (person, address) =&gt; new { PersonName = person.Name, City = address.City });
</code></pre>
<h3 id="java-5">Java</h3>
<pre><code class="language-java">List&lt;Person&gt; persons = List.of(
        new Person(1, &quot;John&quot;),
        new Person(2, &quot;Jane&quot;),
        new Person(3, &quot;Alice&quot;)
);

List&lt;Address&gt; addresses = List.of(
        new Address(1, &quot;New York&quot;),
        new Address(2, &quot;London&quot;),
        new Address(3, &quot;Paris&quot;)
);

List&lt;JoinedData&gt; joinedData = persons.stream()
                                     .join(addresses.stream(),
                                           person -&gt; person.getId(),
                                           address -&gt; address.getId(),
                                           (person, address) -&gt; new JoinedData(person.getName(), address.getCity()))
                                     .collect(Collectors.toList());
</code></pre>
<h3 id="output-5">Output</h3>
<pre><code>[{ PersonName = &quot;John&quot;, City = &quot;New York&quot; },
 { PersonName = &quot;Jane&quot;, City = &quot;London&quot; },
 { PersonName = &quot;Alice&quot;, City = &quot;Paris&quot; }]
</code></pre>
<h2 id="any-anymatch">Any / AnyMatch</h2>
<p>Checks if any element in the sequence satisfies a specified condition.</p>
<h3 id="c-6">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var anyEven = numbers.Any(n =&gt; n % 2 == 0);
</code></pre>
<h3 id="java-6">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
boolean anyEven = numbers.stream()
                         .anyMatch(n -&gt; n % 2 == 0);
</code></pre>
<h3 id="output-6">Output</h3>
<pre><code>true
</code></pre>
<h2 id="all-allmatch">All / AllMatch</h2>
<p>Checks if all elements in the sequence satisfy a specified condition.</p>
<h3 id="c-7">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var allPositive = numbers.All(n =&gt; n &gt; 0);
</code></pre>
<h3 id="java-7">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);

boolean allPositive = numbers.stream()
                             .allMatch(n -&gt; n &gt; 0);
</code></pre>
<h3 id="output-7">Output</h3>
<pre><code>true
</code></pre>
<h2 id="contains-contains">Contains / contains</h2>
<p>Checks if a sequence contains a specific element.</p>
<h3 id="c-8">C#</h3>
<pre><code class="language-csharp">var fruits = new List&lt;string&gt; { &quot;apple&quot;, &quot;banana&quot;, &quot;cherry&quot; };
var containsBanana = fruits.Contains(&quot;banana&quot;);
</code></pre>
<h3 id="java-8">Java</h3>
<pre><code class="language-java">List&lt;String&gt; fruits = Arrays.asList(&quot;apple&quot;, &quot;banana&quot;, &quot;cherry&quot;);
boolean containsBanana = fruits.contains(&quot;banana&quot;);
</code></pre>
<h3 id="output-8">Output</h3>
<pre><code>true
</code></pre>
<h2 id="count-count">Count / count</h2>
<p>Returns the number of elements in a sequence.</p>
<h3 id="c-9">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var count = numbers.Count();
</code></pre>
<h3 id="java-9">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
long count = numbers.stream()
                    .count();
</code></pre>
<h3 id="output-9">Output</h3>
<pre><code>5
</code></pre>
<h2 id="first-findfirst">First / findFirst</h2>
<p>Returns the first element from a sequence, or throws an exception if the sequence is empty.</p>
<h3 id="c-10">C#</h3>
<pre><code class="language-csharp">var = new List&lt;int&gt; { 1, 2, 3, 4, 5 };
var firstNumber = numbers.First();
</code></pre>
<h3 id="java-10">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
int firstNumber = numbers.stream()
                         .findFirst()
                         .orElseThrow(NoSuchElementException::new);
</code></pre>
<h3 id="output-10">Output</h3>
<pre><code>1
</code></pre>
<h2 id="firstordefault-findfirst">FirstOrDefault / findFirst</h2>
<p>Returns the first element from a sequence, or returns the default value for the element type if the sequence is empty.</p>
<h3 id="c-11">C#</h3>
<pre><code class="language-csharp">var numbers = new List&lt;int&gt;();
var firstOrDefaultNumber = numbers.FirstOrDefault();
</code></pre>
<h3 id="java-11">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = new ArrayList&lt;&gt;();
int firstOrDefaultNumber = numbers.stream()
                                  .findFirst()
                                  .orElse(0);
</code></pre>
<h3 id="output-11">Output</h3>
<pre><code>0
</code></pre>
<h2 id="last-reduce">Last / reduce</h2>
<p>Returns the last element from a sequence, or throws an exception if the sequence is empty.</p>
<h3 id="c-12">C#</h3>
<pre><code class="language-csharp">List&lt;int&gt; numbers = new List&lt;int&gt; { 1, 2, 3, 4, 5 };

int lastNumber = numbers.Last();
</code></pre>
<h3 id="java-12">Java</h3>
<pre><code class="language-java">List&lt;Integer&gt; numbers = Arrays.asList(1, 2, 3, 4, 5);
int lastNumber = numbers.stream()
